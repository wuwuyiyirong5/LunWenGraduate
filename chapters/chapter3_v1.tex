Euler~法是求解非线性算子方程的另一种重要的迭代方法.
如果矩阵映射~$f$~由~(\ref{eq:f(X)=0})~定义,
且其~Fr\'{e}chet~导数是可逆的, 那么其一般地迭代格式可写为
\begin{equation*}
X_{k + 1} = X_k - [\IO + L_f(X_k)] \inv{f'(X_k)} f(X_k), \quad k =
0, 1, 2, \ldots,
\end{equation*}
其中算子~$L_f(X)$~为
$$
L_f(X) = \frac{1}{2} \inv{f'(X)} f''(X) \inv{f'(X)} f(X).
$$
%where $\IO$ denotes the identity operator and operator $L_f(X) =
%\frac{1}{2} \inv{f'(X)} f''(X) \inv{f'(X)} f(X)$, can be rewritten
%as follows:
如果初始点~$X_0$~与~$A$~是可交换的且对任意的~$k \geq 0$,
可保证~$X_k$~是非奇异的, 那么可将~Euler~法写成如下简单形式:
\begin{equation}
\label{it:EM} X_{k + 1} = E(X_k), \quad k =0, 1, 2,  \ldots,
\end{equation}
%provided $X_0$ commutes with $A$ and $X_k$ is nonsingular for any $k
%\geq 0$, where
其中
\begin{equation}
\label{it:EM_fun} E(X) = \displaystyle\frac{1}{2p^2} X \left[(2 p^2
- 3 p + 1)\I + 2(2 p - 1)AX^{-p} - (p - 1)\left(
AX^{-p}\right)^2\right],
\end{equation}
%for nonsingular matrix $X \in \CS^{n \times n}$.
而矩阵~$X \in \CS^{n \times n}$~是非奇异的.
说标量~Euler~法~(\ref{it:EM})~意为应用~Euler~法~(\ref{it:EM_BanachSpace})~于标量方程~$f(z)
= z^p-\lambda = 0, z \in \CS$~中所得到的迭代形式
$$
z_{k+1} = E(z_k), \quad k = 0,1,2,\ldots,
$$
其中
$$
E(z) = \frac{1}{2p^2} z \left[(2 p^2 - 3 p + 1) + 2(2 p - 1)\lambda
z^{-p} - (p - 1)\left( \lambda z^{-p}\right)^2\right],
$$
即为~(\ref{it:EM_fun})~的标量形式.

有别于~\cite[定理~3.2]{Cardoso2011}~的证明,
这里我们给出另一种更为简便的证法. 首先, 需要如下的结果:

\begin{lemma}
\label{lem:fun_u(t)}
设复函数~$u_{\textup{E}}(z)$~由~$(\ref{fun:EM_u(z)})$~定义.
则对于任意的~$j \geq 3$~有
%\begin{equation}
%\label{fun:EM_u(z)} u(t) := \frac{1}{1 - \frac{1}{p}t -
%\frac{1}{2}\left(\frac{1}{p} - \frac{1}{p^2}\right)t^2}, \ \ |t| <
%\frac{p}{p - 1}(\sqrt{2p - 1} - 1).
%\end{equation}
%We have
\begin{equation}
\label{eq:up(j)(0)} (u^p_{\textup{E}})^{(j)} (0) = \left(1 + \frac{j
- 1}{p}\right) (u^p_{\textup{E}})^{(j - 1)}(0) + \left(1 + \frac{j
-2}{2p}\right)(j - 1)\left(1 - \frac{1}{p}\right)
(u^p_{\textup{E}})^{(j - 2)}(0)
\end{equation}
且
\begin{equation}
\label{neq:up(j)(0)_j1} (u^p_{\textup{E}})^{(j)}(0) > 0 \quad
\forall \ j \geq 1.
\end{equation}
\end{lemma}

\begin{proof}
由~(\ref{fun:EM_u(z)}),
可得~$u^p_{\textup{E}}(z)$~与其导数有如下的关系:
$$
\left(1 + \left(1 - \frac{1}{p}\right)z\right) u_{\textup{E}}^p(z) =
\left(1 - \frac{1}{p}z - \frac{1}{2}\left(\frac{1}{p} -
\frac{1}{p^2}\right)z^2\right) (u^p_{\textup{E}}(z))'.
$$
对于任意的~$j \geq 3$, 对上式进行~$j -
1$~次微分并应用~Leibnitz~公式有
\begin{align}
\lefteqn{\left(1 + \left(1 -
\frac{1}{p}\right)z\right)(u^p_{\textup{E}})^{(j -
1)}(z)  + (j - 1) \left(1 - \frac{1}{p}\right) (u^p_{\textup{E}})^{(j - 2)} (z)} \nonumber\\
& \quad\qquad\qquad =  \left(1 - \frac{1}{p}z -
\frac{1}{2}\left(\frac{1}{p} -
\frac{1}{p^2}\right)z^2\right) ((u^p_{\textup{E}})')^{(j - 1)}(z) \nonumber \\
& \qquad\qquad\qquad\quad + (j - 1) \left(- \frac{1}{p} -
\left(\frac{1}{p} -
\frac{1}{p^2}\right)z\right) ((u^p_{\textup{E}})')^{(j - 2)} (z)  \nonumber\\
& \qquad\qquad\qquad\quad + \frac{(j - 1)(j - 2)}{2} \left(-
\frac{1}{p} + \frac{1}{p^2}\right)  ((u^p_{\textup{E}})')^{(j - 3)}
(z). \label{eq:up_up'_j-1_times}
\end{align}
于是, 在~(\ref{eq:up_up'_j-1_times})~中令~$z =
0$~即得~(\ref{eq:up(j)(0)}). 而对于~(\ref{neq:up(j)(0)_j1}), 由于
\begin{align*}
(u^p_{\textup{E}})' (z) & = \left(1 + \left(1 -
\frac{1}{p}\right)z\right)
u^{p + 1}_{\textup{E}} (z), \\
(u^p_{\textup{E}})'' (z) & = \left(1 - \frac{1}{p}\right)u^{p +
1}_{\textup{E}}(z) + \frac{p + 1}{p} \left(1 + \left(1 -
\frac{1}{p}\right)z\right)^2  u^{p + 2}_{\textup{E}} (z),
\end{align*}
故
\begin{equation}
\label{eq:up'(0)up''(0)} (u^p_{\textup{E}})'(0) = 1 \ \ \text{and} \
\ (u^p_{\textup{E}})''(0) = 2.
\end{equation}

2)}(0),

或等价地, 不等式

\begin{equation*}
\textup{err}(X) = \frac{\|A - X^p\|}{\|A\|},
\end{equation*}
由该图可知, 当~$p =
2:6$~时耦合~Euler~法~(\ref{it:EM_MatpthRoot_Coupled})~得到了很好的计算结果,
但当~$p = 7:15$~时, 计算结果很变得很差. 因此,
为了能够在计算中提高精度, 我们将采用预处理的方式来改进,
这一点将在第~4~章进行阐述.

\section{关于~Halley~法的注记}



如果初始矩阵~$X_0$~与矩阵~$A$~是可交换的, 且对于任意的~$k \geq 0$,
$$
(p+1)X_k^p + (p-1)A
$$
是非奇异的,
那么计算矩阵~$A$~的主~$p$~次根~$A^{1/p}$~的~Halley~法由如下格式给出:
\begin{equation}
\label{it:HM} X_{k+1} = X_k\left[(p+1)X_k^p + (p-1)A\right]^{-1}
\left[(p-1)X_k^p + (p+1)A\right],\quad k=0,1,2,\ldots,
\end{equation}


类似于~Newton~法~(\ref{it:NM})~和~Euler~法~(\ref{it:EM}),
关于~Halley~法~(\ref{it:HM})~亦可得到相应的收敛性结果. 令
\begin{equation}
\label{fun:HM_phi(z)} \phi_{\textup{H}}(z) := 1 -
u_{\textup{H}}^p(z)(1 - z), \quad z \in \MCD^{\textup{H}}_1,
\end{equation}
其中
\begin{equation*}
u_{\textup{H}}(z) := \frac{1-\displaystyle\frac{p-1}{2p}z}{1 -
\displaystyle\frac{p+1}{2p}z}, \quad z \in \MCD^{\textup{H}}_1,
\end{equation*}
而
\begin{equation}
\label{set:HM_D1} \MCD^{\textup{H}}_1 := \left\{z \in \CS: |z| <
\frac{2p}{p + 1}\right\}.
\end{equation}


定义集合
\begin{equation}
\label{set:HM_R} \MCR_{\textup{H}} := \left\{z\in \CS: 1 - z \in
\overline{\MCD} \bigcup \left(\MCD^{\textup{H}}_1 \bigcap
\MCD^{\textup{H}}_2\right)\right\}
\end{equation}
及
\begin{equation}
\label{set:HM_R_hat} \widehat{\MCR}_{\textup{H}} := \left\{z\in\CS:
1-z \in \MCD \bigcup \left(\MCD^{\textup{H}}_1 \bigcap
\MCD^{\textup{H}}_2\right)\right\},
\end{equation}
其中~$\MCD^{\textup{H}}_1$~由~(\ref{set:HM_D1})~定义,
$\overline{\MCD}$~为~$\MCD$~的闭包, $\MCD$~由~(\ref{set:D})~定义, 而
\begin{equation}
\label{set:HM_D2} \displaystyle \MCD^{\textup{H}}_2  := \left\{z \in
\CS: \sup_{m \geq 3}
\left\{\frac{|S_m^{\textup{H}}(z)|}{|z|}\right\} \cdot \frac{|z|
+|\phi_{\textup{H}}(z)|}{\big||z| - |\phi_{\textup{H}}(z)|\big|} < 1
\right\},
\end{equation}
其中
\begin{equation}
\label{ser:HM_Sm(z)} S_m^{\textup{H}}(z) = \sum_{j = 3}^{m}
c_j^{\textup{H}} z^{j}, \quad z \in \MCD^{\textup{H}}_1,
\end{equation}
$c_j^{\textup{H}} = \phi^{(j)}_{\textup{H}}(0)/j!$,
$\phi_{\textup{H}}$~由~(\ref{fun:HM_phi(z)})~定义.
类似于定理~\ref{th:MatrixNMCon}~和~\ref{th:MatrixEMCon}, 我们有


\begin{theorem}
\label{th:MatrixHMCon}
假设~$\MCR_{\textup{H}}$~和~$\widehat{\MCR}_2$~分别由
~$(\ref{set:HM_R})$~和定义~$(\ref{set:HM_R_hat})$. 如果矩阵~$A \in
\CS^{n \times n}$~的所有特征值都落在~$\MCR_{\textup{H}}$, 其中,
且其所有零特征值都是半单的, 那么由~Halley~法~$(\ref{it:HM})$~以~$X_0
=
\I$~为初始点进行迭代产生的矩阵序列~$\{X_k\}$~收敛于矩阵~$A$~的主~$p$~次根~$A^{1/p}$.
此外, 如果矩阵~$A$~的所有特征值都落在
~$\widehat{\MCR}_2\backslash\{0\}$, 那么收敛速度是三阶的.
\end{theorem}
