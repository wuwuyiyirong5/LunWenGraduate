\chapter{计算矩阵~$p$~次根的~Newton~法}
\label{chapter:NM_MatrixRoot}


%Newton's method and Halley's method are the ones most widely used
%for computing the principal $p$th root $A^{1/p}$. Their iterative
%forms can be obtained by applying the corresponding classical forms
%to the matrix equation

应用迭代法计算给定矩阵~$A \in \CS^{n\times n}$~的~$p$~次根,
实际上是求解如下的非线性矩阵方程
\begin{equation}
\label{eq:f(X)=0} f(X) = X^p - A = 0, \quad p \geq 2.
\end{equation}
%Formally, Newton's method can be written as follows:
若假设初始迭代点~$X_0$~与矩阵~$A$~是可交换的,  且对于任意的~$k \geq
0$, $X_k$~都是非奇异的,
则计算上述矩阵方程的~Newton~法~(\ref{it:NM_BanachSpace})~可写成:
\begin{equation}
\label{it:NM} X_{k+1}= N(X_k),\ \ \ k = 0,1,2,\ldots,
\end{equation}
%provided $X_0$ commutes with $A$ and that $X_k$ is nonsingular for
%any $k \geq 0$, where
其中
\begin{equation}
\label{it:NM_fun} N(X) := \frac{1}{p}X\left[(p-1)\I +
AX^{-p}\right].
\end{equation}
说标量~Newton~法~(\ref{it:NM})~意为应用~Newton~法~(\ref{it:NM_BanachSpace})~于标量方程~$f(z)
= z^p-\lambda = 0, z \in \CS$~中所得到的迭代形式
$$
z_{k+1} = N(z_k), \quad k = 0,1,2,\ldots,
$$
其中
$$
N(z) = \frac{1}{p}z\left[(p-1) + \lambda z^{-p}\right],
$$
即为~(\ref{it:NM_fun})~的标量形式.


本章将研究通过~Newton~法来计算矩阵~$p$~次根的一个新的 收敛性结果.
新的结果比其他现有的结果在计算矩阵~$p$~次根时有更好的计算效果.
本章中总假定整数~$p \geq 2$.

\section{收敛性结果}


在给出本章的主要收敛结果前, 先引入一些记号. 令
\begin{equation}
\label{fun:NM_phi(z)} \phi_{\textup{N}}(z) := 1 -
u^p_{\text{N}}(z)(1 - z), \quad z \in \MCD^{\textup{N}}_1,
\end{equation}
其中
\begin{equation*}
u_{\textup{N}}(z) := \frac{1}{1 - \frac{1}{p}z}, \quad z \in
\MCD^{\textup{N}}_1,
\end{equation*}
及
\begin{equation}
\label{set:NM_D1} \MCD^{\textup{N}}_1 := \left\{z \in \CS: |z| < p
\right\}.
\end{equation}
定义
\begin{equation}
\label{set:NM_R} \MCR_{\textup{N}} := \left\{z\in \CS: 1 - z \in
\overline{\MCD} \bigcup \left(\MCD^{\textup{N}}_1 \bigcap
\MCD^{\textup{N}}_2\right)\right\}
\end{equation}
及
\begin{equation}
\label{set:NM_R_hat} \widehat{\MCR}_{\textup{N}} := \left\{z\in\CS:
1-z \in \MCD \bigcup \left(\MCD^{\textup{N}}_1 \bigcap
\MCD^{\textup{N}}_2\right)\right\},
\end{equation}
其中~$\MCD^{\textup{N}}_1$~由~(\ref{set:NM_D1})~给出, $\MCD$~定义为
\begin{equation}
\label{set:D} \MCD  := \{z \in \CS: |z| < 1\},
\end{equation}
$\overline{\MCD}$~为~$\MCD$~的闭包, 而
\begin{equation}
\label{set:NM_D2} \MCD^{\textup{N}}_2  := \left\{z \in \CS: \sup_{m
\geq 2} \left\{\frac{|S^{\textup{N}}_m(z)|}{|z|}\right\} \cdot
\frac{|z| +|\phi_{\textup{N}}(z)|}{\big||z| -
|\phi_{\textup{N}}(z)|\big|} < 1 \right\},
\end{equation}
其中~$\phi_{\textup{N}}$~ 由~(\ref{fun:NM_phi(z)})~给出,
\begin{equation}
\label{ser:NM_Sm(z)} S^{\textup{N}}_m(z) = \sum\limits_{j = 2}^{m}
c^{\textup{N}}_j z^{j}, \quad z \in \MCD^{\textup{N}}_1,
\end{equation}
这里
\begin{equation*}
\label{cons:NM_c1j} c^{\textup{N}}_j =
\frac{p(p+1)\cdot(p+j-2)}{(j-1)!p^{j-1}} -
\frac{p(p+1)\cdot(p+j-1)}{j!p^j} > 0, \quad j=2,3,\ldots,
\end{equation*}
并满足
$$
\sum_{j=2}^\infty c^{\textup{N}}_j=1.
$$
故 ~$\phi_{\textup{N}}(z)$ 有如下的~Maclaurin~级数形式
\begin{equation}
\label{fun:NM_phi1(z)_series} \phi_{\textup{N}}(z) =
\sum_{j=2}^\infty c^{\textup{N}}_j z^j, \quad z \in
\MCD^{\textup{N}}_1,
\end{equation}
其中 ~$\MCD^{\textup{N}}_1$~ 由 ~(\ref{set:NM_D1})~给出.

基于上述的记号,
下面给出应用~Newton~法~(\ref{it:NM})~来计算矩阵主~$p$~次根时的收敛性结果：


\begin{theorem}
\label{th:MatrixNMCon}
设~$\MCR_{\textup{N}}$~由~$(\ref{set:NM_R})$~定义,
而~$\widehat{\MCR}_{\textup{N}}$~ 由~$(\ref{set:NM_R_hat})$~定义.
如果矩阵~$A \in \CS^{n \times
n}$~的所有特征值都属于~$\MCR_{\textup{N}}$~
且所有零特征值都是半单的, 那么以~$X_0 =
\I$~为初始点的~Newton~法~$(\ref{it:NM})$~迭代所产生的矩阵序列~$\{X_k\}$~收敛于
矩阵~$A$~的主~$p$~次根~$A^{1/p}$. 特别地,
如果~$A$~的所有特征值都属于~$\widehat{\MCR}_{\textup{N}}\backslash\{0\}$,
那么~Newton~法~$(\ref{it:NM})$~是~Q~二阶收敛的.
\end{theorem}

\label{lem:NM_r(N(z))_r(z)}
设~$r(z,\lambda)$~由~$(\ref{fun:NM_r(z)})$~ 定义. 对于某个复数~$z

\begin{equation}

\end{equation}

\label{ineq:NM_abs_r(N(z))_2} |r(N(z),\lambda)| \leq \sup_{m \geq
2}\left\{\left|\frac{S^{\textup{N}}_m(u)}{u^2}\right|\right\}\cdot

|r(z,\lambda)|^2,

其中~$\overline{\MCD}$~为~$\MCD$~的闭包,
而~$\MCD$~由~$(\ref{set:D})$~定义,
$S^{\textup{N}}_m(u)$~由~$(\ref{ser:NM_Sm(z)})$~定义且对于每一个~$u
\in \MCD^{\textup{N}}_1$~均满足~$|u| > |r(z,\lambda)|$, $m \geq 2$.
\end{lemma}

\begin{align*}
\gamma(f,z) & =

\end{align*}

由~$\gamma-$定理~(即定理~\ref{th:SmaleGammaTh})~可立即得到如下结果:

\lambda^{1/p}$~的逼近零点.
\end{theorem}

\end{equation}
下面的结果取自~\cite[定理~7.2]{Wang2000a}~的一维情形:


\begin{theorem}[{\cite[定理~7.2]{Wang2000a}}]

\label{radius:gamma_AZ_ref} \widetilde{\gamma}(f,\zeta) |z-\zeta| <
\frac{2c+1 - \sqrt{c(4c+3)}}{c+1},
\end{equation}
那么~$z$~是~$f$~关于相关零点~$\zeta$~的一个逼近零点.
\end{theorem}

\begin{theorem}

由~(\ref{set:G2})~定义的收敛域~$\MCG_2$~比由~(\ref{set:G1})~定义的~$\MCG_1$~更大.





但需要指出的是, 我们目前仍无法确定~$z=1$~时的最大收敛域. 然而,
当考虑对~$z=1$~作一小扰动时, 相应地收敛域发生了一些较大的变化. 例如,
当~$p$~取固定值~100~时,
图~\ref{fig:reg_gamma_AZ_ref_pert}~给出了不同的小扰动值~$\epsilon$~下保证~$z
= 1 +
\epsilon$~为~$f$~关于相关零点~$\lambda^{1/p}$~的一个逼近零点的~$\lambda
\in \CS$~的区域. 从子图~(e)-(h)~可知, 当~$\epsilon$~取区间~$[0.006,
0.012]$~上的值时可得到较大的收敛域,
这或许将为未来设计更好的算法提供一个观察.
